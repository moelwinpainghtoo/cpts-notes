
Host file generation

```bash
nxc smb 10.10.10.6 --generate-hosts-file /tmp/hosts.tmp && sudo tee -a /etc/hosts < /tmp/hosts.tmp && rm /tmp/hosts.tmp
```

krb5 file

```bash
nxc smb 10.129.85.54 -u 'alex' -p 'lol123!mD' --generate-krb5-file /tmp/krb5.tmp && sudo tee -a /etc/krb5.conf < /tmp/krb5.tmp && rm /tmp/krb5.tmp
```

```bash

# List SMB shares using null session (no creds)
smbclient -N -L //<IP>

# Connect to a specific SMB share
smbclient //<IP>/<share>
smbclient //10.129.85.54/devshare -U 'WINMEDIUM.WINMEDIUM/alex%lol123!mD'

# Download all files from share
prompt OFF
mget *

# Dump users/groups via SAMR (Impacket)
samrdump.py <IP>

# Enumerate shares and permissions
smbmap -H <IP>

# Check accessible shares with null creds
crackmapexec smb <IP> --shares -u '' -p ''

# Full automated SMB enumeration
enum4linux-ng.py <IP> -A

# Run all SMB-related NSE scripts
nmap -p445 --script smb* <IP>

```
